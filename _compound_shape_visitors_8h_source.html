154 {
155  JPH_INLINE CastShapeVisitor(const ShapeCast &inShapeCast, const ShapeCastSettings &inShapeCastSettings, const CompoundShape *inShape, Vec3Arg inScale, const ShapeFilter &inShapeFilter, Mat44Arg inCenterOfMassTransform2, const SubShapeIDCreator &inSubShapeIDCreator1, const SubShapeIDCreator &inSubShapeIDCreator2, CastShapeCollector &ioCollector) :

159  mShapeFilter(inShapeFilter),
160  mCollector(ioCollector),
161  mCenterOfMassTransform2(inCenterOfMassTransform2),
162  mSubShapeIDCreator1(inSubShapeIDCreator1),
163  mSubShapeIDCreator2(inSubShapeIDCreator2)
164  {
165  // Determine ray properties of cast
166  mInvDirection.Set(inShapeCast.mDirection);
167 
168  // Determine bounding box of cast shape
169  mBoxCenter = inShapeCast.mShapeWorldBounds.GetCenter();
170  mBoxExtent = inShapeCast.mShapeWorldBounds.GetExtent();
171 
172  // Determine amount of bits for sub shape
173  mSubShapeBits = inShape->GetSubShapeIDBits();
174  }

180  }

183  JPH_INLINE Vec4 TestBounds(Vec4Arg inBoundsMinX, Vec4Arg inBoundsMinY, Vec4Arg inBoundsMinZ, Vec4Arg inBoundsMaxX, Vec4Arg inBoundsMaxY, Vec4Arg inBoundsMaxZ) const
184  {
185  // Scale the bounding boxes
186  Vec4 bounds_min_x, bounds_min_y, bounds_min_z, bounds_max_x, bounds_max_y, bounds_max_z;
187  AABox4Scale(mScale, inBoundsMinX, inBoundsMinY, inBoundsMinZ, inBoundsMaxX, inBoundsMaxY, inBoundsMaxZ, bounds_min_x, bounds_min_y, bounds_min_z, bounds_max_x, bounds_max_y, bounds_max_z);
188 
189  // Enlarge them by the casted shape's box extents
190  AABox4EnlargeWithExtent(mBoxExtent, bounds_min_x, bounds_min_y, bounds_min_z, bounds_max_x, bounds_max_y, bounds_max_z);
191 
192  // Test ray against the bounding boxes
193  return RayAABox4(mBoxCenter, mInvDirection, bounds_min_x, bounds_min_y, bounds_min_z, bounds_max_x, bounds_max_y, bounds_max_z);

197  JPH_INLINE void VisitShape(const SubShape &inSubShape, uint32 inSubShapeIndex)
198  {
199  JPH_ASSERT(inSubShape.IsValidScale(mScale));
200 
201  // Create ID for sub shape
202  SubShapeIDCreator shape2_sub_shape_id = mSubShapeIDCreator2.PushID(inSubShapeIndex, mSubShapeBits);
203 
204  // Calculate the local transform for this sub shape
205  Mat44 local_transform = Mat44::sRotationTranslation(inSubShape.GetRotation(), mScale * inSubShape.GetPositionCOM());

213  CollisionDispatch::sCastShapeVsShape(shape_cast, mShapeCastSettings, inSubShape.mShape, inSubShape.TransformScale(mScale), mShapeFilter, center_of_mass_transform2, mSubShapeIDCreator1, shape2_sub_shape_id, mCollector);
