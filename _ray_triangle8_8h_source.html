11JPH_INLINE Vec8 RayTriangle8(Vec3Arg inOrigin, Vec3Arg inDirection, Vec8Arg inV0X, Vec8Arg inV0Y, Vec8Arg inV0Z, Vec8Arg inV1X, Vec8Arg inV1Y, Vec8Arg inV1Z, Vec8Arg inV2X, Vec8Arg inV2Y, Vec8Arg inV2Z)
12{
13 // Epsilon
14 Vec8 epsilon = Vec8::sReplicate(1.0e-12f);
15
16 // Zero & one
17 Vec8 zero = Vec8::sZero();
18 Vec8 one = Vec8::sReplicate(1.0f);
19
20 // Find vectors for two edges sharing inV0
21 Vec8 e1x = inV1X - inV0X;
22 Vec8 e1y = inV1Y - inV0Y;
23 Vec8 e1z = inV1Z - inV0Z;
24 Vec8 e2x = inV2X - inV0X;
25 Vec8 e2y = inV2Y - inV0Y;
26 Vec8 e2z = inV2Z - inV0Z;
27
28 // Get direction vector components
29 Vec8 dx = Vec8::sSplatX(Vec4(inDirection));
30 Vec8 dy = Vec8::sSplatY(Vec4(inDirection));
31 Vec8 dz = Vec8::sSplatZ(Vec4(inDirection));
32
33 // Begin calculating determinant - also used to calculate u parameter
34 Vec8 px = dy * e2z - dz * e2y;
35 Vec8 py = dz * e2x - dx * e2z;
36 Vec8 pz = dx * e2y - dy * e2x;
37
38 // if determinant is near zero, ray lies in plane of triangle
39 Vec8 det = e1x * px + e1y * py + e1z * pz;
40
41 // Check which determinants are near zero
42 UVec8 det_near_zero = Vec8::sLess(det.Abs(), epsilon);
43
44 // Set components of the determinant to 1 that are near zero to avoid dividing by zero
45 det = Vec8::sSelect(det, Vec8::sReplicate(1.0f), det_near_zero);
46
47 // Calculate distance from inV0 to ray origin
48 Vec8 sx = Vec8::sSplatX(Vec4(inOrigin)) - inV0X;
49 Vec8 sy = Vec8::sSplatY(Vec4(inOrigin)) - inV0Y;
50 Vec8 sz = Vec8::sSplatZ(Vec4(inOrigin)) - inV0Z;
51
52 // Calculate u parameter and flip sign if determinant was negative
53 Vec8 u = (sx * px + sy * py + sz * pz) / det;
54
55 // Prepare to test v parameter
56 Vec8 qx = sy * e1z - sz * e1y;
57 Vec8 qy = sz * e1x - sx * e1z;
58 Vec8 qz = sx * e1y - sy * e1x;
59
60 // Calculate v parameter and flip sign if determinant was negative
61 Vec8 v = (dx * qx + dy * qy + dz * qz) / det;
62
63 // Get intersection point and flip sign if determinant was negative
64 Vec8 t = (e2x * qx + e2y * qy + e2z * qz) / det;
65
66 // Check if there is an intersection
67 UVec8 no_intersection =
68 UVec8::sOr
69 (
70 UVec8::sOr
71 (
72 UVec8::sOr
73 (
74 det_near_zero,
75 Vec8::sLess(u, zero)
76 ),
77 UVec8::sOr
78 (
79 Vec8::sLess(v, zero),
80 Vec8::sGreater(u + v, one)
81 )
82 ),
83 Vec8::sLess(t, zero)
84 );
85
86 // Select intersection point or FLT_MAX based on if there is an intersection or not
87 return Vec8::sSelect(t, Vec8::sReplicate(FLT_MAX), no_intersection);
88}
