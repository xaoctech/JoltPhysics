19 // Calculate submerged volume * 6 and center of mass * 4 for a tetrahedron with 4 vertices submerged
20 // inV1 .. inV4 are submerged
21 inline static void sTetrahedronVolume4(Vec3Arg inV1, Vec3Arg inV2, Vec3Arg inV3, Vec3Arg inV4, float &outVolumeTimes6, Vec3 &outCenterTimes4)
22 {
23 // Calculate center of mass and mass of this tetrahedron,
24 // see: https://en.wikipedia.org/wiki/Tetrahedron#Volume
25 outVolumeTimes6 = max((inV1 - inV4).Dot((inV2 - inV4).Cross(inV3 - inV4)), 0.0f); // All contributions should be positive because we use a reference point that is on the surface of the hull
26 outCenterTimes4 = inV1 + inV2 + inV3 + inV4;
27 }
28
29 // Get the intersection point with a plane.
30 // inV1 is inD1 distance away from the plane, inV2 is inD2 distance away from the plane
31 inline static Vec3 sGetPlaneIntersection(Vec3Arg inV1, float inD1, Vec3Arg inV2, float inD2)
32 {
33 JPH_ASSERT(Sign(inD1) != Sign(inD2), "Assuming both points are on opposite ends of the plane");
34 float delta = inD1 - inD2;
35 if (abs(delta) < 1.0e-6f)
36 return inV1; // Parallel to plane, just pick a point
37 else
38 return inV1 + inD1 * (inV2 - inV1) / delta;
39 }
40
41 // Calculate submerged volume * 6 and center of mass * 4 for a tetrahedron with 1 vertex submerged
42 // inV1 is submerged, inV2 .. inV4 are not
43 // inD1 .. inD4 are the distances from the points to the plane
44 inline static void sTetrahedronVolume1(Vec3Arg inV1, float inD1, Vec3Arg inV2, float inD2, Vec3Arg inV3, float inD3, Vec3Arg inV4, float inD4, float &outVolumeTimes6, Vec3 &outCenterTimes4)
45 {
46 // A tetrahedron with 1 point submerged is cut along 3 edges forming a new tetrahedron
47 Vec3 v2 = sGetPlaneIntersection(inV1, inD1, inV2, inD2);
48 Vec3 v3 = sGetPlaneIntersection(inV1, inD1, inV3, inD3);
49 Vec3 v4 = sGetPlaneIntersection(inV1, inD1, inV4, inD4);
50
51 #ifdef JPH_DEBUG_RENDERER
52 // Draw intersection between tetrahedron and surface
53 if (Shape::sDrawSubmergedVolumes)
54 {
55 DebugRenderer::sInstance->DrawTriangle(v4, v3, v2, Color::sGreen);
56 DebugRenderer::sInstance->DrawWireTriangle(v4, v3, v2, Color::sWhite);
57 }
58 #endif // JPH_DEBUG_RENDERER
59
60 sTetrahedronVolume4(inV1, v2, v3, v4, outVolumeTimes6, outCenterTimes4);
61 }
62
63 // Calculate submerged volume * 6 and center of mass * 4 for a tetrahedron with 2 vertices submerged
64 // inV1, inV2 are submerged, inV3, inV4 are not
65 // inD1 .. inD4 are the distances from the points to the plane
66 inline static void sTetrahedronVolume2(Vec3Arg inV1, float inD1, Vec3Arg inV2, float inD2, Vec3Arg inV3, float inD3, Vec3Arg inV4, float inD4, float &outVolumeTimes6, Vec3 &outCenterTimes4)
67 {
68 // A tetrahedron with 2 points submerged is cut along 4 edges forming a quad
69 Vec3 c = sGetPlaneIntersection(inV1, inD1, inV3, inD3);
70 Vec3 d = sGetPlaneIntersection(inV1, inD1, inV4, inD4);
71 Vec3 e = sGetPlaneIntersection(inV2, inD2, inV4, inD4);
72 Vec3 f = sGetPlaneIntersection(inV2, inD2, inV3, inD3);
73
74 #ifdef JPH_DEBUG_RENDERER
75 // Draw intersection between tetrahedron and surface
76 if (Shape::sDrawSubmergedVolumes)
77 {
78 DebugRenderer::sInstance->DrawTriangle(c, e, d, Color::sGreen);
79 DebugRenderer::sInstance->DrawTriangle(c, f, e, Color::sGreen);
80 DebugRenderer::sInstance->DrawWireTriangle(c, e, d, Color::sWhite);
81 DebugRenderer::sInstance->DrawWireTriangle(c, f, e, Color::sWhite);
82 }
83 #endif // JPH_DEBUG_RENDERER
84
85 // We pick point c as reference (which is on the cut off surface)
86 // This leaves us with three tetrahedrons to sum up (any faces that are in the same plane as c will have zero volume)
87 Vec3 center1, center2, center3;
88 float volume1, volume2, volume3;
89 sTetrahedronVolume4(e, f, inV2, c, volume1, center1);
90 sTetrahedronVolume4(e, inV1, d, c, volume2, center2);
91 sTetrahedronVolume4(e, inV2, inV1, c, volume3, center3);
92
93 // Tally up the totals
94 outVolumeTimes6 = volume1 + volume2 + volume3;
95 outCenterTimes4 = outVolumeTimes6 > 0.0f? (volume1 * center1 + volume2 * center2 + volume3 * center3) / outVolumeTimes6 : Vec3::sZero();
96 }
97
98 // Calculate submerged volume * 6 and center of mass * 4 for a tetrahedron with 3 vertices submerged
99 // inV1, inV2, inV3 are submerged, inV4 is not
100 // inD1 .. inD4 are the distances from the points to the plane
101 inline static void sTetrahedronVolume3(Vec3Arg inV1, float inD1, Vec3Arg inV2, float inD2, Vec3Arg inV3, float inD3, Vec3Arg inV4, float inD4, float &outVolumeTimes6, Vec3 &outCenterTimes4)
102 {
103 // A tetrahedron with 1 point above the surface is cut along 3 edges forming a new tetrahedron
104 Vec3 v1 = sGetPlaneIntersection(inV1, inD1, inV4, inD4);
105 Vec3 v2 = sGetPlaneIntersection(inV2, inD2, inV4, inD4);
106 Vec3 v3 = sGetPlaneIntersection(inV3, inD3, inV4, inD4);
107
108 #ifdef JPH_DEBUG_RENDERER
109 // Draw intersection between tetrahedron and surface
110 if (Shape::sDrawSubmergedVolumes)
111 {
112 DebugRenderer::sInstance->DrawTriangle(v3, v2, v1, Color::sGreen);
113 DebugRenderer::sInstance->DrawWireTriangle(v3, v2, v1, Color::sWhite);
114 }
115 #endif // JPH_DEBUG_RENDERER
116
117 Vec3 dry_center, total_center;
118 float dry_volume, total_volume;
119
120 // We first calculate the part that is above the surface
121 sTetrahedronVolume4(v1, v2, v3, inV4, dry_volume, dry_center);
122
123 // Calculate the total volume
124 sTetrahedronVolume4(inV1, inV2, inV3, inV4, total_volume, total_center);
125
126 // From this we can calculate the center and volume of the submerged part
127 outVolumeTimes6 = max(total_volume - dry_volume, 0.0f);
128 outCenterTimes4 = outVolumeTimes6 > 0.0f? (total_center * total_volume - dry_center * dry_volume) / outVolumeTimes6 : Vec3::sZero();
129 }

148 PolyhedronSubmergedVolumeCalculator(const Mat44 &inTransform, const Vec3 *inPoints, int inPointStride, int inNumPoints, const Plane &inSurface, Point *ioBuffer) :
149 mPoints(ioBuffer)
150 {
151 // Convert the points to world space and determine the distance to the surface
152 float reference_dist = FLT_MAX;
153 for (int p = 0; p < inNumPoints; ++p)
154 {
155 // Calculate values
156 Vec3 transformed_point = inTransform * *reinterpret_cast<const Vec3 *>(reinterpret_cast<const uint8 *>(inPoints) + p * inPointStride);
157 float dist = inSurface.SignedDistance(transformed_point);
158 bool above = dist >= 0.0f;
159
160 // Keep track if all are above or below
161 mAllAbove &= above;
162 mAllBelow &= !above;
163
164 // Calculate lowest point, we use this to create tetrahedrons out of all faces
165 if (reference_dist > dist)
166 {
167 mReferencePointIdx = p;
168 reference_dist = dist;
169 }
170
171 // Store values
172 ioBuffer->mPosition = transformed_point;
173 ioBuffer->mDistanceToSurface = dist;
174 ioBuffer->mAboveSurface = above;
175 ++ioBuffer;
176 }
177 }

198 void AddFace(int inIdx1, int inIdx2, int inIdx3)
199 {
200 JPH_ASSERT(inIdx1 != mReferencePointIdx && inIdx2 != mReferencePointIdx && inIdx3 != mReferencePointIdx, "A face using the reference point will not contribute to the volume");
201
202 // Find the points
203 const Point &ref = mPoints[mReferencePointIdx];
204 const Point &p1 = mPoints[inIdx1];
205 const Point &p2 = mPoints[inIdx2];
206 const Point &p3 = mPoints[inIdx3];
207
208 // Determine which vertices are submerged
209 uint code = (p1.mAboveSurface? 0 : 0b001) | (p2.mAboveSurface? 0 : 0b010) | (p3.mAboveSurface? 0 : 0b100);
210
211 float volume;
212 Vec3 center;
213 switch (code)
214 {
215 case 0b000:
216 // One point submerged
217 sTetrahedronVolume1(ref.mPosition, ref.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, volume, center);
218 break;
219
220 case 0b001:
221 // Two points submerged
222 sTetrahedronVolume2(ref.mPosition, ref.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, volume, center);
223 break;
224
225 case 0b010:
226 // Two points submerged
227 sTetrahedronVolume2(ref.mPosition, ref.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, volume, center);
228 break;
229
230 case 0b100:
231 // Two points submerged
232 sTetrahedronVolume2(ref.mPosition, ref.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, volume, center);
233 break;
234
235 case 0b011:
236 // Three points submerged
237 sTetrahedronVolume3(ref.mPosition, ref.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, volume, center);
238 break;
239
240 case 0b101:
241 // Three points submerged
242 sTetrahedronVolume3(ref.mPosition, ref.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, volume, center);
243 break;
244
245 case 0b110:
246 // Three points submerged
247 sTetrahedronVolume3(ref.mPosition, ref.mDistanceToSurface, p3.mPosition, p3.mDistanceToSurface, p2.mPosition, p2.mDistanceToSurface, p1.mPosition, p1.mDistanceToSurface, volume, center);
248 break;
249
250 case 0b111:
251 // Four points submerged
252 sTetrahedronVolume4(ref.mPosition, p3.mPosition, p2.mPosition, p1.mPosition, volume, center);
253 break;
254
255 default:
256 // Should not be possible
257 JPH_ASSERT(false);
258 volume = 0.0f;
259 center = Vec3::sZero();
260 break;
261 }
262
263 mSubmergedVolume += volume;
264 mCenterOfBuoyancy += volume * center;
265 }
266
268 void GetResult(float &outSubmergedVolume, Vec3 &outCenterOfBuoyancy) const
269 {
270 outCenterOfBuoyancy = mSubmergedVolume > 0.0f? mCenterOfBuoyancy / (4.0f * mSubmergedVolume) : Vec3::sZero(); // Do this before dividing submerged volume by 6 to get correct weight factor
271 outSubmergedVolume = mSubmergedVolume / 6.0f;
272 }

285 // Aggregator for submerged volume and center of buoyancy

287 Vec3 mCenterOfBuoyancy = Vec3::sZero();
