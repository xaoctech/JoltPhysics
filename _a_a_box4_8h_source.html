12JPH_INLINE UVec4 AABox4VsBox(const AABox &inBox1, Vec4Arg inBox2MinX, Vec4Arg inBox2MinY, Vec4Arg inBox2MinZ, Vec4Arg inBox2MaxX, Vec4Arg inBox2MaxY, Vec4Arg inBox2MaxZ)
13{
14 // Splat values of box 1
15 Vec4 box1_minx = inBox1.mMin.SplatX();
16 Vec4 box1_miny = inBox1.mMin.SplatY();
17 Vec4 box1_minz = inBox1.mMin.SplatZ();
18 Vec4 box1_maxx = inBox1.mMax.SplatX();
19 Vec4 box1_maxy = inBox1.mMax.SplatY();
20 Vec4 box1_maxz = inBox1.mMax.SplatZ();
21
22 // Test separation over each axis
23 UVec4 nooverlapx = UVec4::sOr(Vec4::sGreater(box1_minx, inBox2MaxX), Vec4::sGreater(inBox2MinX, box1_maxx));
24 UVec4 nooverlapy = UVec4::sOr(Vec4::sGreater(box1_miny, inBox2MaxY), Vec4::sGreater(inBox2MinY, box1_maxy));
25 UVec4 nooverlapz = UVec4::sOr(Vec4::sGreater(box1_minz, inBox2MaxZ), Vec4::sGreater(inBox2MinZ, box1_maxz));
26
27 // Return overlap
28 return UVec4::sNot(UVec4::sOr(UVec4::sOr(nooverlapx, nooverlapy), nooverlapz));
29}

32JPH_INLINE void AABox4Scale(Vec3Arg inScale, Vec4Arg inBoxMinX, Vec4Arg inBoxMinY, Vec4Arg inBoxMinZ, Vec4Arg inBoxMaxX, Vec4Arg inBoxMaxY, Vec4Arg inBoxMaxZ, Vec4 &outBoundsMinX, Vec4 &outBoundsMinY, Vec4 &outBoundsMinZ, Vec4 &outBoundsMaxX, Vec4 &outBoundsMaxY, Vec4 &outBoundsMaxZ)
33{

35 Vec4 scaled_min_x = scale_x * inBoxMinX;
36 Vec4 scaled_max_x = scale_x * inBoxMaxX;

38 outBoundsMaxX = Vec4::sMax(scaled_min_x, scaled_max_x);

41 Vec4 scaled_min_y = scale_y * inBoxMinY;

43 outBoundsMinY = Vec4::sMin(scaled_min_y, scaled_max_y);
44 outBoundsMaxY = Vec4::sMax(scaled_min_y, scaled_max_y);

47 Vec4 scaled_min_z = scale_z * inBoxMinZ;
48 Vec4 scaled_max_z = scale_z * inBoxMaxZ;
49 outBoundsMinZ = Vec4::sMin(scaled_min_z, scaled_max_z);
50 outBoundsMaxZ = Vec4::sMax(scaled_min_z, scaled_max_z);

70JPH_INLINE UVec4 AABox4VsPoint(Vec3Arg inPoint, Vec4Arg inBoxMinX, Vec4Arg inBoxMinY, Vec4Arg inBoxMinZ, Vec4Arg inBoxMaxX, Vec4Arg inBoxMaxY, Vec4Arg inBoxMaxZ)
71{
72 // Splat point to 4 component vectors
73 Vec4 point_x = Vec4(inPoint).SplatX();
74 Vec4 point_y = Vec4(inPoint).SplatY();
75 Vec4 point_z = Vec4(inPoint).SplatZ();
76
77 // Test if point overlaps with box
78 UVec4 overlapx = UVec4::sAnd(Vec4::sGreaterOrEqual(point_x, inBoxMinX), Vec4::sLessOrEqual(point_x, inBoxMaxX));
79 UVec4 overlapy = UVec4::sAnd(Vec4::sGreaterOrEqual(point_y, inBoxMinY), Vec4::sLessOrEqual(point_y, inBoxMaxY));
80 UVec4 overlapz = UVec4::sAnd(Vec4::sGreaterOrEqual(point_z, inBoxMinZ), Vec4::sLessOrEqual(point_z, inBoxMaxZ));
81
82 // Test if all are overlapping
83 return UVec4::sAnd(UVec4::sAnd(overlapx, overlapy), overlapz);
84}

87JPH_INLINE UVec4 AABox4VsBox(Mat44Arg inOrientation, Vec3Arg inHalfExtents, Vec4Arg inBoxMinX, Vec4Arg inBoxMinY, Vec4Arg inBoxMinZ, Vec4Arg inBoxMaxX, Vec4Arg inBoxMaxY, Vec4Arg inBoxMaxZ, float inEpsilon = 1.0e-6f)
88{
89 // Taken from: Real Time Collision Detection - Christer Ericson
90 // Chapter 4.4.1, page 103-105.
91 // Note that the code is swapped around: A is the aabox and B is the oriented box (this saves us from having to invert the orientation of the oriented box)
92
93 // Compute translation vector t (the translation of B in the space of A)
94 Vec4 t[3] {
95 inOrientation.GetTranslation().SplatX() - 0.5f * (inBoxMinX + inBoxMaxX),
96 inOrientation.GetTranslation().SplatY() - 0.5f * (inBoxMinY + inBoxMaxY),
97 inOrientation.GetTranslation().SplatZ() - 0.5f * (inBoxMinZ + inBoxMaxZ) };
98
99 // Compute common subexpressions. Add in an epsilon term to
100 // counteract arithmetic errors when two edges are parallel and
101 // their cross product is (near) null (see text for details)
102 Vec3 epsilon = Vec3::sReplicate(inEpsilon);
103 Vec3 abs_r[3] { inOrientation.GetAxisX().Abs() + epsilon, inOrientation.GetAxisY().Abs() + epsilon, inOrientation.GetAxisZ().Abs() + epsilon };
104
105 // Half extents for a
106 Vec4 a_half_extents[3] {
107 0.5f * (inBoxMaxX - inBoxMinX),
108 0.5f * (inBoxMaxY - inBoxMinY),
109 0.5f * (inBoxMaxZ - inBoxMinZ) };
110
111 // Half extents of b
112 Vec4 b_half_extents_x = inHalfExtents.SplatX();
113 Vec4 b_half_extents_y = inHalfExtents.SplatY();
114 Vec4 b_half_extents_z = inHalfExtents.SplatZ();
115
116 // Each component corresponds to 1 overlapping OBB vs ABB
117 UVec4 overlaps = UVec4(0xffffffff, 0xffffffff, 0xffffffff, 0xffffffff);
118
119 // Test axes L = A0, L = A1, L = A2
120 Vec4 ra, rb;
121 for (int i = 0; i < 3; i++)
122 {
123 ra = a_half_extents[i];
124 rb = b_half_extents_x * abs_r[0][i] + b_half_extents_y * abs_r[1][i] + b_half_extents_z * abs_r[2][i];
125 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual(t[i].Abs(), ra + rb));
126 }
127
128 // Test axes L = B0, L = B1, L = B2
129 for (int i = 0; i < 3; i++)
130 {
131 ra = a_half_extents[0] * abs_r[i][0] + a_half_extents[1] * abs_r[i][1] + a_half_extents[2] * abs_r[i][2];
132 rb = Vec4::sReplicate(inHalfExtents[i]);
133 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[0] * inOrientation(0, i) + t[1] * inOrientation(1, i) + t[2] * inOrientation(2, i)).Abs(), ra + rb));
134 }
135
136 // Test axis L = A0 x B0
137 ra = a_half_extents[1] * abs_r[0][2] + a_half_extents[2] * abs_r[0][1];
138 rb = b_half_extents_y * abs_r[2][0] + b_half_extents_z * abs_r[1][0];
139 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[2] * inOrientation(1, 0) - t[1] * inOrientation(2, 0)).Abs(), ra + rb));
140
141 // Test axis L = A0 x B1
142 ra = a_half_extents[1] * abs_r[1][2] + a_half_extents[2] * abs_r[1][1];
143 rb = b_half_extents_x * abs_r[2][0] + b_half_extents_z * abs_r[0][0];
144 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[2] * inOrientation(1, 1) - t[1] * inOrientation(2, 1)).Abs(), ra + rb));
145
146 // Test axis L = A0 x B2
147 ra = a_half_extents[1] * abs_r[2][2] + a_half_extents[2] * abs_r[2][1];
148 rb = b_half_extents_x * abs_r[1][0] + b_half_extents_y * abs_r[0][0];
149 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[2] * inOrientation(1, 2) - t[1] * inOrientation(2, 2)).Abs(), ra + rb));
150
151 // Test axis L = A1 x B0
152 ra = a_half_extents[0] * abs_r[0][2] + a_half_extents[2] * abs_r[0][0];
153 rb = b_half_extents_y * abs_r[2][1] + b_half_extents_z * abs_r[1][1];
154 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[0] * inOrientation(2, 0) - t[2] * inOrientation(0, 0)).Abs(), ra + rb));
155
156 // Test axis L = A1 x B1
157 ra = a_half_extents[0] * abs_r[1][2] + a_half_extents[2] * abs_r[1][0];
158 rb = b_half_extents_x * abs_r[2][1] + b_half_extents_z * abs_r[0][1];
159 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[0] * inOrientation(2, 1) - t[2] * inOrientation(0, 1)).Abs(), ra + rb));
160
161 // Test axis L = A1 x B2
162 ra = a_half_extents[0] * abs_r[2][2] + a_half_extents[2] * abs_r[2][0];
163 rb = b_half_extents_x * abs_r[1][1] + b_half_extents_y * abs_r[0][1];
164 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[0] * inOrientation(2, 2) - t[2] * inOrientation(0, 2)).Abs(), ra + rb));
165
166 // Test axis L = A2 x B0
167 ra = a_half_extents[0] * abs_r[0][1] + a_half_extents[1] * abs_r[0][0];
168 rb = b_half_extents_y * abs_r[2][2] + b_half_extents_z * abs_r[1][2];
169 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[1] * inOrientation(0, 0) - t[0] * inOrientation(1, 0)).Abs(), ra + rb));
170
171 // Test axis L = A2 x B1
172 ra = a_half_extents[0] * abs_r[1][1] + a_half_extents[1] * abs_r[1][0];
173 rb = b_half_extents_x * abs_r[2][2] + b_half_extents_z * abs_r[0][2];
174 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[1] * inOrientation(0, 1) - t[0] * inOrientation(1, 1)).Abs(), ra + rb));
175
176 // Test axis L = A2 x B2
177 ra = a_half_extents[0] * abs_r[2][1] + a_half_extents[1] * abs_r[2][0];
178 rb = b_half_extents_x * abs_r[1][2] + b_half_extents_y * abs_r[0][2];
179 overlaps = UVec4::sAnd(overlaps, Vec4::sLessOrEqual((t[1] * inOrientation(0, 2) - t[0] * inOrientation(1, 2)).Abs(), ra + rb));
180
181 // Return if the OBB vs AABBs are intersecting
182 return overlaps;
183}

192JPH_INLINE UVec4 AABox4VsSphere(Vec4Arg inCenterX, Vec4Arg inCenterY, Vec4Arg inCenterZ, Vec4Arg inRadiusSq, Vec4Arg inBoxMinX, Vec4Arg inBoxMinY, Vec4Arg inBoxMinZ, Vec4Arg inBoxMaxX, Vec4Arg inBoxMaxY, Vec4Arg inBoxMaxZ)
193{
194 // Get closest point on box
195 Vec4 closest_x = Vec4::sMin(Vec4::sMax(inCenterX, inBoxMinX), inBoxMaxX);
196 Vec4 closest_y = Vec4::sMin(Vec4::sMax(inCenterY, inBoxMinY), inBoxMaxY);
197 Vec4 closest_z = Vec4::sMin(Vec4::sMax(inCenterZ, inBoxMinZ), inBoxMaxZ);
198
199 // Test the distance from the center of the sphere to the box is smaller than the radius
200 Vec4 distance_sq = Square(closest_x - inCenterX) + Square(closest_y - inCenterY) + Square(closest_z - inCenterZ);
201 return Vec4::sLessOrEqual(distance_sq, inRadiusSq);
202}
