43 template <EMotionType Type1, EMotionType Type2>
44 JPH_INLINE bool ApplyVelocityStep(MotionProperties *ioMotionProperties1, MotionProperties *ioMotionProperties2, Vec3Arg inWorldSpaceAxis, float inLambda) const
45 {
46 // Apply impulse if delta is not zero
47 if (inLambda != 0.0f)
48 {
49 // Calculate velocity change due to constraint
50 //
51 // Impulse:
52 // P = J^T lambda
53 //
54 // Euler velocity integration:
55 // v' = v + M^-1 P
56 if constexpr (Type1 == EMotionType::Dynamic)
57 {
58 ioMotionProperties1->SubLinearVelocityStep((inLambda * ioMotionProperties1->GetInverseMass()) * inWorldSpaceAxis);
59 ioMotionProperties1->SubAngularVelocityStep(inLambda * Vec3::sLoadFloat3Unsafe(mInvI1_R1PlusUxAxis));
60 }
61 if constexpr (Type2 == EMotionType::Dynamic)
62 {
63 ioMotionProperties2->AddLinearVelocityStep((inLambda * ioMotionProperties2->GetInverseMass()) * inWorldSpaceAxis);
64 ioMotionProperties2->AddAngularVelocityStep(inLambda * Vec3::sLoadFloat3Unsafe(mInvI2_R2xAxis));
65 }
66 return true;
67 }
68
69 return false;
70 }

74 template <EMotionType Type1, EMotionType Type2>
75 JPH_INLINE void TemplatedCalculateConstraintProperties(float inDeltaTime, const MotionProperties *inMotionProperties1, Mat44Arg inInvI1, Vec3Arg inR1PlusU, const MotionProperties *inMotionProperties2, Mat44Arg inInvI2, Vec3Arg inR2, Vec3Arg inWorldSpaceAxis, float inBias = 0.0f, float inC = 0.0f, float inFrequency = 0.0f, float inDamping = 0.0f)
76 {
77 JPH_ASSERT(inWorldSpaceAxis.IsNormalized(1.0e-5f));
78
79 // Calculate properties used below
80 Vec3 r1_plus_u_x_axis;
81 if constexpr (Type1 != EMotionType::Static)
82 {
83 r1_plus_u_x_axis = inR1PlusU.Cross(inWorldSpaceAxis);
84 r1_plus_u_x_axis.StoreFloat3(&mR1PlusUxAxis);
85 }
86 else
87 JPH_IF_DEBUG(Vec3::sNaN().StoreFloat3(&mR1PlusUxAxis));
88
89 Vec3 r2_x_axis;
90 if constexpr (Type2 != EMotionType::Static)
91 {
92 r2_x_axis = inR2.Cross(inWorldSpaceAxis);
93 r2_x_axis.StoreFloat3(&mR2xAxis);
94 }
95 else
96 JPH_IF_DEBUG(Vec3::sNaN().StoreFloat3(&mR2xAxis));
97
98 // Calculate inverse effective mass: K = J M^-1 J^T
99 float inv_effective_mass;
100
101 if constexpr (Type1 == EMotionType::Dynamic)
102 {
103 Vec3 invi1_r1_plus_u_x_axis = inInvI1 * r1_plus_u_x_axis;
104 invi1_r1_plus_u_x_axis.StoreFloat3(&mInvI1_R1PlusUxAxis);
105 inv_effective_mass = inMotionProperties1->GetInverseMass() + invi1_r1_plus_u_x_axis.Dot(r1_plus_u_x_axis);
106 }
107 else
108 {
109 (void)r1_plus_u_x_axis; // Fix compiler warning: Not using this (it's not calculated either)
110 JPH_IF_DEBUG(Vec3::sNaN().StoreFloat3(&mInvI1_R1PlusUxAxis);)
111 inv_effective_mass = 0.0f;
112 }
113
114 if constexpr (Type2 == EMotionType::Dynamic)
115 {
116 Vec3 invi2_r2_x_axis = inInvI2 * r2_x_axis;
117 invi2_r2_x_axis.StoreFloat3(&mInvI2_R2xAxis);
118 inv_effective_mass += inMotionProperties2->GetInverseMass() + invi2_r2_x_axis.Dot(r2_x_axis);
119 }
120 else
121 {
122 (void)r2_x_axis; // Fix compiler warning: Not using this (it's not calculated either)
123 JPH_IF_DEBUG(Vec3::sNaN().StoreFloat3(&mInvI2_R2xAxis);)
124 }
125
126 // Calculate effective mass and spring properties
127 mSpringPart.CalculateSpringProperties(inDeltaTime, inv_effective_mass, inBias, inC, inFrequency, inDamping, mEffectiveMass);
128 }

239 template <EMotionType Type1, EMotionType Type2>
240 inline bool TemplatedSolveVelocityConstraint(MotionProperties *ioMotionProperties1, MotionProperties *ioMotionProperties2, Vec3Arg inWorldSpaceAxis, float inMinLambda, float inMaxLambda)
241 {
242 // Calculate jacobian multiplied by linear velocity
243 float jv;
244 if constexpr (Type1 != EMotionType::Static && Type2 != EMotionType::Static)
245 jv = inWorldSpaceAxis.Dot(ioMotionProperties1->GetLinearVelocity() - ioMotionProperties2->GetLinearVelocity());
246 else if constexpr (Type1 != EMotionType::Static)
247 jv = inWorldSpaceAxis.Dot(ioMotionProperties1->GetLinearVelocity());
248 else if constexpr (Type2 != EMotionType::Static)
249 jv = inWorldSpaceAxis.Dot(-ioMotionProperties2->GetLinearVelocity());
250 else
251 JPH_ASSERT(false); // Static vs static is nonsensical!
252
253 // Calculate jacobian multiplied by angular velocity
254 if constexpr (Type1 != EMotionType::Static)
255 jv += Vec3::sLoadFloat3Unsafe(mR1PlusUxAxis).Dot(ioMotionProperties1->GetAngularVelocity());
256 if constexpr (Type2 != EMotionType::Static)
257 jv -= Vec3::sLoadFloat3Unsafe(mR2xAxis).Dot(ioMotionProperties2->GetAngularVelocity());
258
259 // Lagrange multiplier is:
260 //
261 // lambda = -K^-1 (J v + b)
262 float lambda = mEffectiveMass * (jv - mSpringPart.GetBias(mTotalLambda));
263 float new_lambda = Clamp(mTotalLambda + lambda, inMinLambda, inMaxLambda); // Clamp impulse
264 lambda = new_lambda - mTotalLambda; // Lambda potentially got clamped, calculate the new impulse to apply
265 mTotalLambda = new_lambda; // Store accumulated impulse
266
267 return ApplyVelocityStep<Type1, Type2>(ioMotionProperties1, ioMotionProperties2, inWorldSpaceAxis, lambda);
268 }

276 inline bool SolveVelocityConstraint(Body &ioBody1, Body &ioBody2, Vec3Arg inWorldSpaceAxis, float inMinLambda, float inMaxLambda)
277 {
278 EMotionType motion_type1 = ioBody1.GetMotionType();
279 MotionProperties *motion_properties1 = ioBody1.GetMotionPropertiesUnchecked();
280
281 EMotionType motion_type2 = ioBody2.GetMotionType();
282 MotionProperties *motion_properties2 = ioBody2.GetMotionPropertiesUnchecked();
283
284 // Dispatch to the correct templated form
285 switch (motion_type1)
286 {
287 case EMotionType::Dynamic:
288 switch (motion_type2)
289 {
290 case EMotionType::Dynamic:
291 return TemplatedSolveVelocityConstraint<EMotionType::Dynamic, EMotionType::Dynamic>(motion_properties1, motion_properties2, inWorldSpaceAxis, inMinLambda, inMaxLambda);
292
293 case EMotionType::Kinematic:
294 return TemplatedSolveVelocityConstraint<EMotionType::Dynamic, EMotionType::Kinematic>(motion_properties1, motion_properties2, inWorldSpaceAxis, inMinLambda, inMaxLambda);
295
296 case EMotionType::Static:
297 return TemplatedSolveVelocityConstraint<EMotionType::Dynamic, EMotionType::Static>(motion_properties1, motion_properties2, inWorldSpaceAxis, inMinLambda, inMaxLambda);
298
299 default:
300 JPH_ASSERT(false);
301 break;
302 }
303 break;
304
305 case EMotionType::Kinematic:
306 JPH_ASSERT(motion_type2 == EMotionType::Dynamic);
307 return TemplatedSolveVelocityConstraint<EMotionType::Kinematic, EMotionType::Dynamic>(motion_properties1, motion_properties2, inWorldSpaceAxis, inMinLambda, inMaxLambda);
308
309 case EMotionType::Static:
310 JPH_ASSERT(motion_type2 == EMotionType::Dynamic);
311 return TemplatedSolveVelocityConstraint<EMotionType::Static, EMotionType::Dynamic>(motion_properties1, motion_properties2, inWorldSpaceAxis, inMinLambda, inMaxLambda);
312
313 default:
314 JPH_ASSERT(false);
315 break;
316 }
317
318 return false;
319 }

327 inline bool SolvePositionConstraint(Body &ioBody1, Body &ioBody2, Vec3Arg inWorldSpaceAxis, float inC, float inBaumgarte)
328 {
329 // Only apply position constraint when the constraint is hard, otherwise the velocity bias will fix the constraint
330 if (inC != 0.0f && !mSpringPart.IsActive())
331 {
332 // Calculate lagrange multiplier (lambda) for Baumgarte stabilization:
333 //
334 // lambda = -K^-1 * beta / dt * C
335 //
336 // We should divide by inDeltaTime, but we should multiply by inDeltaTime in the Euler step below so they're cancelled out
337 float lambda = -mEffectiveMass * inBaumgarte * inC;
338
339 // Directly integrate velocity change for one time step
340 //
341 // Euler velocity integration:
342 // dv = M^-1 P
343 //
344 // Impulse:
345 // P = J^T lambda
346 //
347 // Euler position integration:
348 // x' = x + dv * dt
349 //
350 // Note we don't accumulate velocities for the stabilization. This is using the approach described in 'Modeling and
351 // Solving Constraints' by Erin Catto presented at GDC 2007. On slide 78 it is suggested to split up the Baumgarte
352 // stabilization for positional drift so that it does not actually add to the momentum. We combine an Euler velocity
353 // integrate + a position integrate and then discard the velocity change.
354 if (ioBody1.IsDynamic())
355 {
356 ioBody1.SubPositionStep((lambda * ioBody1.GetMotionPropertiesUnchecked()->GetInverseMass()) * inWorldSpaceAxis);
357 ioBody1.SubRotationStep(lambda * Vec3::sLoadFloat3Unsafe(mInvI1_R1PlusUxAxis));
358 }
359 if (ioBody2.IsDynamic())
360 {
361 ioBody2.AddPositionStep((lambda * ioBody2.GetMotionPropertiesUnchecked()->GetInverseMass()) * inWorldSpaceAxis);
362 ioBody2.AddRotationStep(lambda * Vec3::sLoadFloat3Unsafe(mInvI2_R2xAxis));
363 }
364 return true;
365 }
366
367 return false;
368 }
