24inline float CenterAngleAroundZero(float inV)
25{

94inline uint CountTrailingZeros(uint32 inValue)
95{
96#if defined(JPH_CPU_X64)
97 #ifdef JPH_USE_LZCNT
98 return _tzcnt_u32(inValue);
