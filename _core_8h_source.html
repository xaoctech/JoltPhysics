195// Macro to enable floating point precise mode and to disable fused multiply add instructions

197 // In clang it appears you cannot turn off -ffast-math and -ffp-contract=fast for a code block
198 // There is #pragma clang fp contract (off) but that doesn't seem to work under clang 9 & 10 when -ffast-math is specified on the commandline (you override it to turn it on, but not off)
199 // There is #pragma float_control(precise, on) but that doesn't work under clang 9.
200 // So for now we compile clang without -ffast-math so the macros are empty
201 #define JPH_PRECISE_MATH_ON
202 #define JPH_PRECISE_MATH_OFF
203#elif defined(_MSC_VER)
204 // Unfortunately there is no way to push the state of fp_contract, so we have to assume it was turned on before JPH_PRECISE_MATH_ON
205 #define JPH_PRECISE_MATH_ON \
206 __pragma(float_control(precise, on, push)) \
207 __pragma(fp_contract(off))
208 #define JPH_PRECISE_MATH_OFF \
209 __pragma(fp_contract(on)) \
210 __pragma(float_control(pop))
