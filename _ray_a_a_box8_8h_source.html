13JPH_INLINE Vec8 RayAABox8(Vec3Arg inOrigin, const RayInvDirection &inInvDirection, Vec8Arg inBoundsMinX, Vec8Arg inBoundsMinY, Vec8Arg inBoundsMinZ, Vec8Arg inBoundsMaxX, Vec8Arg inBoundsMaxY, Vec8Arg inBoundsMaxZ)
14{
15 // Constants
16 Vec8 flt_min = Vec8::sReplicate(-FLT_MAX);
17 Vec8 flt_max = Vec8::sReplicate(FLT_MAX);
18
19 // Origin
20 Vec8 originx = Vec8::sSplatX(Vec4(inOrigin));
21 Vec8 originy = Vec8::sSplatY(Vec4(inOrigin));
22 Vec8 originz = Vec8::sSplatZ(Vec4(inOrigin));
23
24 // Parallel
25 UVec8 parallelx = UVec8::sSplatX(inInvDirection.mIsParallel);
26 UVec8 parallely = UVec8::sSplatY(inInvDirection.mIsParallel);
27 UVec8 parallelz = UVec8::sSplatZ(inInvDirection.mIsParallel);
28
29 // Inverse direction
30 Vec8 invdirx = Vec8::sSplatX(Vec4(inInvDirection.mInvDirection));
31 Vec8 invdiry = Vec8::sSplatY(Vec4(inInvDirection.mInvDirection));
32 Vec8 invdirz = Vec8::sSplatZ(Vec4(inInvDirection.mInvDirection));
33
34 // Test against all three axii simultaneously.
35 Vec8 t1x = (inBoundsMinX - originx) * invdirx;
36 Vec8 t1y = (inBoundsMinY - originy) * invdiry;
37 Vec8 t1z = (inBoundsMinZ - originz) * invdirz;
38 Vec8 t2x = (inBoundsMaxX - originx) * invdirx;
39 Vec8 t2y = (inBoundsMaxY - originy) * invdiry;
40 Vec8 t2z = (inBoundsMaxZ - originz) * invdirz;
41
42 // Compute the max of min(t1,t2) and the min of max(t1,t2) ensuring we don't
43 // use the results from any directions parallel to the slab.
44 Vec8 t_minx = Vec8::sSelect(Vec8::sMin(t1x, t2x), flt_min, parallelx);
45 Vec8 t_miny = Vec8::sSelect(Vec8::sMin(t1y, t2y), flt_min, parallely);
46 Vec8 t_minz = Vec8::sSelect(Vec8::sMin(t1z, t2z), flt_min, parallelz);
47 Vec8 t_maxx = Vec8::sSelect(Vec8::sMax(t1x, t2x), flt_max, parallelx);
48 Vec8 t_maxy = Vec8::sSelect(Vec8::sMax(t1y, t2y), flt_max, parallely);
49 Vec8 t_maxz = Vec8::sSelect(Vec8::sMax(t1z, t2z), flt_max, parallelz);
50
51 // t_min.xyz = maximum(t_min.x, t_min.y, t_min.z);
52 Vec8 t_min = Vec8::sMax(Vec8::sMax(t_minx, t_miny), t_minz);
53
54 // t_max.xyz = minimum(t_max.x, t_max.y, t_max.z);
55 Vec8 t_max = Vec8::sMin(Vec8::sMin(t_maxx, t_maxy), t_maxz);
56
57 // if (t_min > t_max) return FLT_MAX;
58 UVec8 no_intersection = Vec8::sGreater(t_min, t_max);
59
60 // if (t_max < 0.0f) return FLT_MAX;
61 no_intersection = UVec8::sOr(no_intersection, Vec8::sLess(t_max, Vec8::sZero()));
62
63 // if bounds are invalid return FLOAT_MAX;
64 UVec8 bounds_invalid = UVec8::sOr(UVec8::sOr(Vec8::sGreater(inBoundsMinX, inBoundsMaxX), Vec8::sGreater(inBoundsMinY, inBoundsMaxY)), Vec8::sGreater(inBoundsMinZ, inBoundsMaxZ));
65 no_intersection = UVec8::sOr(no_intersection, bounds_invalid);
66
67 // if (inInvDirection.mIsParallel && !(Min <= inOrigin && inOrigin <= Max)) return FLT_MAX; else return t_min;
68 UVec8 no_parallel_overlapx = UVec8::sAnd(parallelx, UVec8::sOr(Vec8::sLess(originx, inBoundsMinX), Vec8::sGreater(originx, inBoundsMaxX)));
69 UVec8 no_parallel_overlapy = UVec8::sAnd(parallely, UVec8::sOr(Vec8::sLess(originy, inBoundsMinY), Vec8::sGreater(originy, inBoundsMaxY)));
70 UVec8 no_parallel_overlapz = UVec8::sAnd(parallelz, UVec8::sOr(Vec8::sLess(originz, inBoundsMinZ), Vec8::sGreater(originz, inBoundsMaxZ)));
71 no_intersection = UVec8::sOr(no_intersection, UVec8::sOr(UVec8::sOr(no_parallel_overlapx, no_parallel_overlapy), no_parallel_overlapz));
72 return Vec8::sSelect(t_min, flt_max, no_intersection);
73}
